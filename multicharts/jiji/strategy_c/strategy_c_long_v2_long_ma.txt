// BTCUSDT 43min
inputs: Enabled(1), OrderSize(10);

if Enabled = 0 then
	#Return;


vars: MaxOpenPositionProfit(0);
if MarketPosition <> 0 then begin
	if MaxOpenPositionProfit < OpenPositionProfit then
		MaxOpenPositionProfit = OpenPositionProfit;
end else begin
	MaxOpenPositionProfit = 0;
end;


if MarketPosition = 0 then begin
	vars: BuyCond(True);
	BuyCond = True;
	BuyCond = BuyCond and Close > Open;
	BuyCond = BuyCond and Close > AverageFC(Close, 6);
	BuyCond = BuyCond and Close > AverageFC(Close, 60);
	BuyCond = BuyCond and Ticks = Highest(Ticks, 3);
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;
end else begin

	if OpenPositionProfit > 420 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.6 then begin
		Sell ("LX_Pullback") all contracts next bar market;
	end;

	if OpenPositionProfit > 360 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.5 then begin
		Sell ("LX_Pullback2") all contracts next bar market;
	end;

	if OpenPositionProfit > 150 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.2 then begin
		Sell ("LX_Pullback3") all contracts next bar market;
	end;

	if OpenPositionProfit > 100 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.1 then begin
		Sell ("LX_Pullback4") all contracts next bar market;
	end;

	if OpenPositionProfit < -1010 * BigPointValue * OrderSize then begin
		Sell ("LX_StopLoss") all contracts next bar market;
	end;

end;
