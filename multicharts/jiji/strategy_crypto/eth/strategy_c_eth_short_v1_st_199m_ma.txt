// ETH 199 min
// Use bar 80
// For swing trading
// Focus on balancing profits and losses.

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: MaxOpenPositionProfit(0);
if MarketPosition <> 0 then begin
	if MaxOpenPositionProfit < OpenPositionProfit then
		MaxOpenPositionProfit = OpenPositionProfit;
end else begin
	MaxOpenPositionProfit = 0;
end;

vars: ATR(0);
ATR = AvgTrueRange(14);

vars: OpenCond(True);
OpenCond = True;

if MarketPosition = 0 then begin
	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and Close < Open;
	ShortCond = ShortCond and Close < AverageFC(Close, 60);
	ShortCond = ShortCond and Close = Lowest(Close, 7);
	ShortCond = ShortCond and Ticks = Highest(Ticks, 5);
	if ShortCond then begin
		SellShort ( "SE" ) 30 * OrderSize contracts next bar market;
	end;

end else begin

	if OpenPositionProfit > 2 * ATR * BigPointValue * OrderSize then begin
		BuyToCover ("SX_Win") all contracts next bar market;
	end;

	if OpenPositionProfit > 0 and OpenPositionProfit < MaxOpenPositionProfit - 3 * ATR * BigPointValue * OrderSize then begin
		BuyToCover ("SX_Pullback") all contracts next bar market;
	end;

	if OpenPositionProfit < -13 * ATR * BigPointValue * OrderSize then begin
		BuyToCover ("SX_StopLoss") all contracts next bar market;
	end;
end;
