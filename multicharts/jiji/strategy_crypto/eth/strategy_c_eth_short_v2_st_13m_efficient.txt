// BTC 13 min
// Use bar 80
// For swing trading
// Focus on balancing profits and losses.

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: OpenCond(True);

if MarketPosition = 0 then begin
	vars: Len(80), PointSum(0), count(0);
	count = Len - 1;
	PointSum = Summation(AbsValue(Close - Close[1]), count) ;
	vars: Efficient(0);
	if PointSum <> 0 then begin
		Efficient = (Close - Close[count]) / PointSum;
	end else begin
		Efficient = 0;
	end;

	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and crosses_below(Efficient, -0.32);
	if ShortCond then begin
		SellShort ( "SE" ) OrderSize contracts next bar market;
	end;

end else begin
	vars: CoverCond(True);
	CoverCond = True;
	CoverCond = CoverCond and LowestBar(Low, Len) > Len * 0.65;
	if CoverCond then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
