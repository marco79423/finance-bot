// TXF 43min
inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: OpenPrice(Open);
if 0845 <= Time and Time <= 0850 then begin
	OpenPrice = Open;
end;

vars: MaxOpenPositionProfit(0);
if MarketPosition <> 0 then begin
	if MaxOpenPositionProfit < OpenPositionProfit then
		MaxOpenPositionProfit = OpenPositionProfit;
end else begin
	MaxOpenPositionProfit = 0;
end;

if MarketPosition = 0 then begin
	vars: ShortCond(True);
	ShortCond = True;
	ShortCond = ShortCond and not (OverMonth and 1300 < Time and Time < 1400);
	ShortCond = ShortCond and not (0845 <= Time and Time <= 0847);
	ShortCond = ShortCond and Close < Open;
	ShortCond = ShortCond and Close < AverageFC(Close, 70);
	ShortCond = ShortCond and Close = Lowest(Close, 3);
	ShortCond = ShortCond and Ticks = Highest(Ticks, 15);
	if ShortCond then
		SellShort ( "SE" ) OrderSize contracts next bar market;

end else begin
	if OverMonth and 1300 < Time and Time < 1400 then begin
		BuyToCover ("SX_Settle") all contracts next bar market;
	end;

	if OpenPositionProfit > 270 * BigPointValue * OrderSize then
		BuyToCover ("SX_MaxWin") all contracts next bar market;

	if OpenPositionProfit > 200 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.98 then begin
		BuyToCover ("SX_Pullback") all contracts next bar market;
	end;

	if OpenPositionProfit > 110 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.88 then begin
		BuyToCover ("SX_Pullback2") all contracts next bar market;
	end;

	if OpenPositionProfit > 90 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.86 then begin
		BuyToCover ("SX_Pullback3") all contracts next bar market;
	end;

	if OpenPositionProfit > 10 * BigPointValue * OrderSize and OpenPositionProfit < MaxOpenPositionProfit * 0.18 then begin
		BuyToCover ("SX_Pullback4") all contracts next bar market;
	end;

	if BarsSinceEntry > 10 then begin
		BuyToCover ("SX_Giveup") all contracts next bar market;
	end;

	if OpenPositionProfit < -70 * BigPointValue * OrderSize then begin
		BuyToCover ("SX_StopLoss") all contracts next bar market;
	end;
end;
