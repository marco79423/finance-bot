inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: FastMALen(50), SlowMALen(170), LookBackLen(3), ThresholdCount(2);
vars: FastMA(0), SlowMA(0);

FastMA = Average(Close, FastMALen);
SlowMA = Average(Close, SlowMALen);

if CountIF(FastMA > SlowMA, LookBackLen) > ThresholdCount then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

if MarketPosition = 1 then begin
	if CountIF(FastMA < SlowMA, LookBackLen) > ThresholdCount then begin
		Sell ("LX") all contracts next bar market;
	end;
end;
