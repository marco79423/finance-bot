// Backtest time: 2014/01/01 - 2024/01/01 Exchange
// Compliance: 1


inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: MALength(5), KBar(325), BBandMultiplier(1.7), ADXMax(22);

vars: MA(0);
MA = Average(Close, MALength);

vars: BBandDown(0), BBandUp(0), BBandMid(0);
BBandDown = BollingerBand(Close, KBar, -BBandMultiplier);
BBandUp = BollingerBand(Close, KBar, BBandMultiplier);
BBandMid = BollingerBand(Close, KBar, 0);


if ADX(14) < ADXMax then begin
	if Close > MA and Close < BBandDown then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;

	if Close < MA and Close > BBandUp then begin
		SellShort ("SE") OrderSize contracts next bar market;
	end;
end;


if MarketPosition = 1 then begin
	if Close > BBandMid then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;

if MarketPosition = -1 then begin
	if Close < BBandMid then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
