inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: Len(5), NN(3.1);
vars: BigK(0);
BigK = StandardDev(Close, Len, 1) * NN ;

if Close > AvgPrice[1] + BigK then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

vars: PullBack_Percent(0.014);

if MarketPosition <> 0 then begin
	if Close < HighSinceEntry - EntryPrice * PullBack_Percent then begin
		Sell ("LX") all contracts next bar market;
	end;
end;
