inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: StdDevLength(165), StdDevMultiplier(2.61), PullBackPercent(0.01);

vars: BigK(0);
BigK = StandardDev(Close, StdDevLength, 1) * StdDevMultiplier;

if MarketPosition <= 0 then begin
	Buy ("LE") OrderSize contracts next bar AvgPrice + BigK stop;
end;

if MarketPosition >= 0 then begin
	SellShort ("SE") OrderSize contracts next bar AvgPrice - BigK stop;
end;

if MarketPosition = 1 then begin
	Sell ("LX") all contracts next bar HighSinceEntry - EntryPrice * PullBackPercent stop;
end;

if MarketPosition = -1 then begin
	BuyToCover ("SX") all contracts next bar LowSinceEntry + EntryPrice * PullBackPercent stop;
end;
