// JNM 20 min
// Backtest time: 2014/01/01 - 2024/01/01 Exchange

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: Len(6), NN(2.8);
vars: BigK(0);
BigK = StandardDev(Close, Len, 1) * NN ;

if Close > AvgPrice[1] + BigK then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

vars: ATR(0);
ATR = AvgTrueRange(14);

vars: ProfitMultiplier(19), StopLossMultiplier(12);

if MarketPosition = 1 then begin
	if OpenPositionProfit > 0 and Close > EntryPrice + ProfitMultiplier * ATR  then begin
		Sell ("LX_PT") all contracts next bar market;
	end;

	if Close < EntryPrice - StopLossMultiplier * ATR then begin
		Sell ("LX_ST") all contracts next bar market;
	end;
end;
