// MYM 30 min
// Use bar 300
// https://hiskio.com/courses/396/lectures/20618

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: HighSinceEntry(-999999);
if MarketPosition <> 0 then begin
	if MarketPosition <> MarketPosition [1] or High > HighSinceEntry then
		HighSinceEntry = High;
end else begin
	HighSinceEntry = -999999;
end;

vars: LowSinceEntry(999999);
if MarketPosition <> 0 then begin
	if MarketPosition <> MarketPosition [1] or Low < LowSinceEntry then
		LowSinceEntry = Low;
end else begin
	LowSinceEntry = -999999;
end;


vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MYM");

vars: Len(10), NN(2.05), BigK(0);
BigK = StandardDev(Close, Len, 1) * NN ;
if BigK < 10 then begin
	BigK = 10;
end;

if MarketPosition = 0 then begin
	vars: BuyCond(True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and Close > AvgPrice[1] + BigK;
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;
end;

if MarketPosition = 1 then begin
	if IsSettlementDay("MYM") then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	if Close < HighSinceEntry - 15.5 * BigK then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;
