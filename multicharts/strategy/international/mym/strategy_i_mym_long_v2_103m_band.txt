// MYM 103 min
// Use bar 40
// For swing trading
// Focus on balancing profits and losses.

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: MaxOpenPositionProfit(0);
if MarketPosition <> 0 then begin
	if MaxOpenPositionProfit < OpenPositionProfit then
		MaxOpenPositionProfit = OpenPositionProfit;
end else begin
	MaxOpenPositionProfit = 0;
end;

var: EntryHigh(0);
if MarketPosition = 1 then begin
	if High > EntryHigh then
		EntryHigh = High;
end else begin
	EntryHigh = 0;
end;

vars: ATR(0);
ATR = AvgTrueRange(14);

vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MYM");

var: KBar(40), BBandAverage(0), BBandDown(0);
BBandAverage = Average(Close, KBar);
BBandDown = BBandAverage - 1.4 * StandardDev(Close, KBar, 1);

var: HH(99999);
if D > D[1] then begin
	HH = OpenD(0) + ATR * 0.3;
end;

vars: OutPrice(0);

if MarketPosition = 0 then begin

	vars: BuyCond(True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and Close cross over BBandDown;
	BuyCond = BuyCond and Close > Close[1];
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
		OutPrice = BBandAverage;
    end;

end else begin
	if IsSettlementDay("MYM") then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	if Close > OutPrice then begin
		Sell ("LX_Out") all contracts next bar market;
	end;

	if BarsSinceEntry > 5 and OpenPositionProfit > 0 then begin
		Sell ("LX_GiveUp") all contracts next bar market;
	end;

	if OpenPositionProfit < -15 * ATR * OrderSize then begin
		Sell ("LX_StopLoss") all contracts next bar market;
	end;

end;
