// Backtest time: 2020/03/01 - 2024/01/01 Exchange

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: LongMALen(19), ShortMALen(15), Len(13), Count(4);

vars: LongMA(0), ShortMA(0);
LongMA = Average(Close, LongMALen);
ShortMA = Average(Close, ShortMALen);

vars: BuyCond(True);
BuyCond = LongMA > ShortMA;
BuyCond = BuyCond and CountIF(Close > LongMA, Len) > Count;
if BuyCond then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

vars: SellCond(True);
SellCond = LongMA < ShortMA;
SellCond = SellCond and CountIF(Close < LongMA, Len) > Count;
if SellCond then begin
	Sell ("LX") all contracts next bar market;
end;
