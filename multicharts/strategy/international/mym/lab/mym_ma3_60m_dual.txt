// MYM 60 min
// Backtest time: 2020/03/01 - 2024/01/01 Exchange
// Use bar 80

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: ShortMALen(10), LongMALen(20), Len(20), Count(17);

vars: OpenCond(True);
OpenCond = True;

vars: ShortMA(0), LongMA(0);
ShortMA = Average(Close, ShortMALen);
LongMA = Average(Close, LongMALen);

vars: BuyCond(True);
BuyCond = OpenCond;
BuyCond = BuyCond and CountIF(ShortMA > LongMA, Len) > Count;
if BuyCond then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

vars: ShortCond(True);
ShortCond = OpenCond;
ShortCond = ShortCond and CountIF(ShortMA < LongMA, Len) > Count;
if ShortCond then begin
	SellShort ("SE") OrderSize contracts next bar market;
end;
