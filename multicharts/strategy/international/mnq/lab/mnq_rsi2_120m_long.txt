// MNQ 120 min
// Backtest time: 2020/03/01 - 2024/01/01 Exchange


inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: RSILen(35), RSIEntryThreshold(48), RSIExitThreshold(52);

vars: RSIValue(0);
RSIValue = RSI(Close, RSILen);

if RSIValue cross over RSIEntryThreshold then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

if RSIValue cross under RSIExitThreshold then begin
	Sell ("LX") all contracts next bar market;
end;
