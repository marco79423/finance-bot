// MNQ 240 min
// Use bar 30
// For swing trading
// Focus on balancing profits and losses.

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: MaxOpenPositionProfit(0);
if MarketPosition <> 0 then begin
	if MaxOpenPositionProfit < OpenPositionProfit then
		MaxOpenPositionProfit = OpenPositionProfit;
end else begin
	MaxOpenPositionProfit = 0;
end;

vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MNQ");

inputs: C1(70), C2(12), C3(17);
if MarketPosition = 0 then begin
	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and Close < Open;
	ShortCond = ShortCond and Close < AverageFC(Close, 10);
	ShortCond = ShortCond and Close = Lowest(Close, 1);
	ShortCond = ShortCond and Ticks = Highest(Ticks, 7);
	if ShortCond then begin
		SellShort ( "SE" ) OrderSize contracts next bar market;
	end;

end else begin
	if IsSettlementDay("MNQ") then begin
		BuyToCover ("SX_Settle") all contracts next bar market;
	end;

	vars: ATR(0);
	ATR = AvgTrueRange(30);

	if OpenPositionProfit > 5 * ATR * OrderSize then begin
		BuyToCover ("SX_Win") all contracts next bar market;
	end;

	if OpenPositionProfit < -2.5 * ATR * OrderSize then begin
		BuyToCover ("SX_StopLoss") all contracts next bar market;
	end;
end;
