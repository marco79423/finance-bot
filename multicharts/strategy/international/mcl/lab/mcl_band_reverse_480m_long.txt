// MCL 480 min
// Use bar 12

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MCL");

vars: KBar(12), BBandDownMultiplier(1.1);
vars: BBandAverage(0), BBandDown(0);
BBandAverage = Average(Close, KBar);
BBandDown = BBandAverage - BBandDownMultiplier * StandardDev(Close, KBar, 1);


if MarketPosition = 0 then begin
	vars: BuyCond (True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and Close cross over BBandDown;
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;
end else begin
	if IsSettlementDay("MCL") then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	if Close > BBandAverage then begin
		Sell ("LX_Out") all contracts next bar market;
	end;

	if Close < BBandDown then begin
		Sell ("LX_StopLoss") all contracts next bar market;
	end;
end;
