// MCL 120 min
// Use bar 45

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MCL");

if MarketPosition = 0 then begin
	vars: MALen(45), TickLen(7);

	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and Close < AverageFC(Close, MALen);
	ShortCond = ShortCond and Ticks = Highest(Ticks, TickLen);
	if ShortCond then begin
		SellShort ( "SE" ) OrderSize contracts next bar market;
	end;
end else begin
	if IsSettlementDay("MCL") then begin
        BuyToCover ("LX_Settle") all contracts next bar market;
    end;

	vars: Profit_Percent(0.01), StopLoss_Percent(0.04);

	if Close < EntryPrice - EntryPrice * Profit_Percent then begin
		BuyToCover ("SX_PT") all contracts next bar market;
	end;

	if High > EntryPrice + EntryPrice * StopLoss_Percent then begin
		BuyToCover ("SX_SL") all contracts next bar market;
	end;

end;
