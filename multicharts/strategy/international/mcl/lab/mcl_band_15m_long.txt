// MCL 15 min
// Use bar 125

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: KBar(125), BBandUpLen(2.4);
vars: BBandUp(0);
BBandUp = BollingerBand(Close, KBar, BBandUpLen);

vars: OpenCond(True);
OpenCond = True;
OpenCond = not IsSettlementDay("MCL");

if MarketPosition = 0 then begin
	vars: BuyCond (True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and Close > BBandUp;
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;
end else begin
	if IsSettlementDay("MCL") then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	vars: Profit_Percent(0.03), PullBack_Percent(0.01), StopLoss_Percent(0.03);

	if Close > EntryPrice + EntryPrice * Profit_Percent then begin
		Sell ("LX_PT") all contracts next bar market;
	end;

	if OpenPositionProfit > 0 and Close < HighSinceEntry - EntryPrice * PullBack_Percent then begin
		Sell ("LX_PullBack") all contracts next bar market;
	end;

	if Low < EntryPrice - StopLoss_Percent * EntryPrice then begin
		Sell ("LX_SL") all contracts next bar market;
	end;

end;
