// Backtest time: 2015/09/01 - 2024/01/01 Exchange
// Compliance: 1


inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: FastMALen(3), SlowMALen(60), LookBackLen(9), ThresholdCount(6);
vars: FastMA(0), SlowMA(0);

FastMA = Average(Close, FastMALen);
SlowMA = Average(Close, SlowMALen);

if FastMA < SlowMA and CountIF(Close > SlowMA, LookBackLen) > ThresholdCount then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

if FastMA > SlowMA and CountIF(Close < SlowMA, LookBackLen) > ThresholdCount then begin
	SellShort ("SE") OrderSize contracts next bar market;
end;
