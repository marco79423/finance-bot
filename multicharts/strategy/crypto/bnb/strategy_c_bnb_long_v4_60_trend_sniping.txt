// BNB 60 min
// Use bar 30

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

var: MA(0);
MA = Average(Close, 30);

vars: OpenCond(True);
OpenCond = True;

vars: RevisedOrderSize(OrderSize);
if Round(6000 / Close, 0) > 1 then begin
	RevisedOrderSize = Round(6000 / Close, 0) * OrderSize;
end else begin
	RevisedOrderSize = OrderSize;
end;

if MarketPosition = 0 then begin
	vars: BuyCond(True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and CountIf(Close > MA, 3 + 2) >= 2;
	BuyCond = BuyCond and CountIf(Close < MA, 5) > 0;
	BuyCond = BuyCond and High > High[1];
	if BuyCond then begin
		Buy ("LE") RevisedOrderSize contracts next bar market;
	end;
end else begin
	vars: SellCond(True);
	SellCond = True;
	SellCond = SellCond and CountIf(Close < MA, 1 + 2) >= 2;
	SellCond = SellCond and CountIf(Close > MA, 1) > 0;
	SellCond = SellCond and Low < Low[1];
	if SellCond then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;
