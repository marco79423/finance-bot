// BNB 60 min
// Use bar 30

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

var: MA(0);
MA = Average(Close, 30);

vars: OpenCond(True);
OpenCond = True;

vars: RevisedOrderSize(OrderSize);
if Round(6000 / Close, 0) > 1 then begin
	RevisedOrderSize = Round(6000 / Close, 0) * OrderSize;
end else begin
	RevisedOrderSize = OrderSize;
end;

if MarketPosition = 0 then begin
	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and CountIf(Close < MA, 3 + 3) >= 3;
	ShortCond = ShortCond and CountIf(Close > MA, 2) > 0;
	ShortCond = ShortCond and Low < Low[1];
	if ShortCond then begin
		SellShort ( "SE" ) RevisedOrderSize contracts next bar market;
	end;
end else begin
	vars: CoverCond(True);
	CoverCond = True;
	CoverCond = CoverCond and CountIf(Close < MA, 4 + 5) >= 5;
	CoverCond = CoverCond and CountIf(Close > MA, 2) > 0;
	CoverCond = CoverCond and High > High[1];
	if CoverCond then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
