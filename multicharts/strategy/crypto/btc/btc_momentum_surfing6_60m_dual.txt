// BTC 60 min
// Backtest time: 2019/10/01 - 2024/01/01 +00:00

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: RevisedOrderSize(OrderSize);
if Round(100000 / Close, 0) > 1 then begin
	RevisedOrderSize = Round(100000 / Close, 0) * OrderSize;
end else begin
	RevisedOrderSize = OrderSize;
end;


vars: Len(20), NN(7.3);
vars: BigK(0);
BigK = StandardDev(Close, Len, 1) * NN;

if MarketPosition <= 0 then begin
	Buy ("LE") RevisedOrderSize contracts next bar AvgPrice + BigK stop;
end;

if MarketPosition >= 0 then begin
	SellShort ("SE") RevisedOrderSize contracts next bar AvgPrice - BigK stop;
end;

vars: PullBackPercent(0.31);

if MarketPosition = 1 then begin
	Sell ("LX_PT") all contracts next bar HighSinceEntry - EntryPrice * PullBackPercent stop;
end;

if MarketPosition = -1 then begin
	BuyToCover ("SX_PT") all contracts next bar LowSinceEntry + EntryPrice * PullBackPercent stop;
end;
