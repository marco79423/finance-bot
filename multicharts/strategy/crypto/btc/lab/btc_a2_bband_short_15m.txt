// Backtest time: 2019/10/01 - 2024/01/01 +00:00
// Compliance: 1

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: RevisedOrderSize(OrderSize);
if Round(100000 / Close, 0) > 1 then begin
	RevisedOrderSize = Round(100000 / Close, 0) * OrderSize;
end else begin
	RevisedOrderSize = OrderSize;
end;

vars: KBar(65), BBandMultiplier(4.5);

vars: BBandDown(0), BBandMid(0);
BBandDown = BollingerBand(Close, KBar, -BBandMultiplier);
BBandMid = BollingerBand(Close, KBar, 0);

if Close < BBandDown then begin
	SellShort ("SE") RevisedOrderSize contracts next bar market;
end;

if MarketPosition = -1 then begin
	if Close > BBandMid then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
