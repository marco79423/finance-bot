// BTC 60 min
// Backtest time: 2019/10/01 - 2024/01/01 +00:00
// Compliance: 1

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: RevisedOrderSize(OrderSize);
if Round(100000 / Close, 0) > 1 then begin
	RevisedOrderSize = Round(100000 / Close, 0) * OrderSize;
end else begin
	RevisedOrderSize = OrderSize;
end;

vars: ADXMax(24), StdDevLength(6), StdDevMultiplier(5.3), KBar(135), BBandMultiplier(0.9);

vars: BigK(0);
BigK = StandardDev(Close, StdDevLength, 1) * StdDevMultiplier;

vars: BBandUp(0), BBandDown(0), BBandMid(0);
BBandUp = BollingerBand(Close, KBar, BBandMultiplier);
BBandDown = BollingerBand(Close, KBar, -BBandMultiplier);
BBandMid = BollingerBand(Close, KBar, 0);

if MarketPosition = 0 and ADX(14) < ADXMax then begin
	if Close < BBandDown then begin
		Buy ("LE") OrderSize contracts next bar Close + BigK stop;
	end;

	if Close > BBandUp then begin
		SellShort ("SE") OrderSize contracts next bar Close - BigK stop;
	end;
end;

if MarketPosition = 1 then begin
	if Close > BBandMid then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;

if MarketPosition = -1 then begin
	if Close < BBandMid then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
