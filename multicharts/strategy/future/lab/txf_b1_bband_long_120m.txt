// TXF 120 min
// Backtest time: 2018/01/01 - 2024/01/01 Exchange
// Compliance: 1

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;



vars: StdDevLength(25), StdDevMultiplier(1.1), KBar(90), BBandMultiplier(1.3);

vars: BigK(0);
BigK = StandardDev(Close, StdDevLength, 1) * StdDevMultiplier;

vars: BBandDown(0), BBandMid(0);
BBandDown = BollingerBand(Close, KBar, -BBandMultiplier);
BBandMid = BollingerBand(Close, KBar, 0);

vars: OpenCond(True);
OpenCond = not (IsTXFSettlementDay and 1200 < Time and Time < 1400);
OpenCond = OpenCond and not (0845 <= Time and Time <= 0847);
OpenCond = OpenCond and not (1500 <= Time and Time <= 1502);

if MarketPosition = 0 then begin
	if OpenCond and Close < BBandDown then begin
		Buy ("LE") OrderSize contracts next bar Close + BigK stop;
	end;
end;

if MarketPosition = 1 then begin
	if Close > BBandMid then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;
