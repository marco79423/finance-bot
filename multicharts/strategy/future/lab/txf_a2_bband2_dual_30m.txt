// TXF 30 min
// Backtest time: 2018/01/01 - 2014/01/01 Exchange
// Fee (12 + 5) * 20
// Compliance: 1


inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: KBar(50), BBandMultiplier(3.5), PullBackPercent(0.034);

vars: BBandUp(0), BBandDown(0);
BBandUp = BollingerBand(Close, KBar, BBandMultiplier);
BBandDown = BollingerBand(Close, KBar, -BBandMultiplier);

if Close > BBandUp then begin
	Buy ("LE") OrderSize contracts next bar market;
end;

if Close < BBandDown then begin
	SellShort ( "SE" ) OrderSize contracts next bar market;
end;

if MarketPosition = 1 then begin
	Sell ("LX") all contracts next bar HighSinceEntry - EntryPrice * PullBackPercent stop;
end;

if MarketPosition = -1 then begin
	BuyToCover ("SX") all contracts next bar LowSinceEntry + EntryPrice * PullBackPercent stop;
end;
