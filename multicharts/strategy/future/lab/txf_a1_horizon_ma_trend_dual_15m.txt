// TXF 15 min
// Backtest time: 2018/01/01 - 2024/01/01 Exchange
// Compliance: 1

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;


vars: FastMALen(40), SlowMALen(170), LongFilterMALen(1150);

vars: FastMA(0), SlowMA(0), LongFilterMA(0);
FastMA = Average(Close, FastMALen);
SlowMA = Average(Close, SlowMALen);
LongFilterMA = Average(Close, LongFilterMALen);


vars: OpenCond(True);
OpenCond = not (IsTXFSettlementDay and 1200 < Time and Time < 1400);
OpenCond = OpenCond and not (0845 <= Time and Time <= 0847);
OpenCond = OpenCond and not (1500 <= Time and Time <= 1502);


if MarketPosition <= 0 then begin
	if OpenCond and FastMA > SlowMA and Close > LongFilterMA then begin
		Buy ( "LE" ) OrderSize contracts next bar market;
	end;
end;

if MarketPosition >= 0 then begin
	if OpenCond and FastMA < SlowMA and Close < LongFilterMA then begin
		SellShort ( "SE" ) OrderSize contracts next bar market;
	end;
end;
