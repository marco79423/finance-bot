// TXF 5 min
// Use bar 60
// For swing trading
// Focus on balancing profits and losses.

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

vars: OpenCond(True);
OpenCond = not (IsTXFSettlementDay and 1200 < Time and Time < 1400);
OpenCond = OpenCond and not (0845 <= Time and Time <= 0847);
OpenCond = OpenCond and not (1500 <= Time and Time <= 1502);

vars: Len(60), PointSum(0), count(0);
count = Len - 1;
PointSum = Summation(AbsValue(Close - Close[1]), count) ;
vars: Efficient(0);
if PointSum <> 0 then begin
	Efficient = (Close - Close[count]) / PointSum;
end else begin
	Efficient = 0;
end;

if MarketPosition <= 0 then begin
	vars: BuyCond(True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and crosses_above(Efficient, 0.09);
	if BuyCond then begin
		Buy ( "LE" ) OrderSize contracts next bar market;
	end;
end;

if MarketPosition >= 0 then begin
	vars: ShortCond(True);
	ShortCond = OpenCond;
	ShortCond = ShortCond and crosses_below(Efficient, -0.37);
	if ShortCond then begin
		SellShort ( "SE" ) OrderSize contracts next bar market;
	end;
end;

if MarketPosition = 1 then begin
	if IsTXFSettlementDay and 1200 < Time and Time < 1400 then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	vars: SellCond(True);
	SellCond = True;
	SellCond = SellCond and HighestBar(Low, Len) > Len * 3.7;
	if SellCond then begin
		Sell ("LX_Out") all contracts next bar market;
	end;
end;

if MarketPosition = -1 then begin
	if IsTXFSettlementDay and 1200 < Time and Time < 1400 then begin
		BuyToCover ("SX_Settle") all contracts next bar market;
	end;

	vars: CoverCond(True);
	CoverCond = True;
	CoverCond = CoverCond and LowestBar(High, Len) > Len * 0.8;
	if CoverCond then begin
		BuyToCover ("SX_Out") all contracts next bar market;
	end;
end;
