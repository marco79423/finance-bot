// TXF 60 min
// Use bar 30

inputs: Enabled(1), OrderSize(1);

if Enabled = 0 then
	#Return;

var: MA(0);
MA = Average(Close, 10);

vars: OpenCond(True);
OpenCond = True;
OpenCond = not (IsTXFSettlementDay and 1200 < Time and Time < 1400);
OpenCond = OpenCond and not (0845 <= Time and Time <= 0847);
OpenCond = OpenCond and not (1500 <= Time and Time <= 1502);

if MarketPosition = 0 then begin
	vars: BuyCond(True);
	BuyCond = OpenCond;
	BuyCond = BuyCond and CountIf(Close > MA, 5 + 5) >= 5;
	BuyCond = BuyCond and CountIf(Close < MA, 5) > 0;
	BuyCond = BuyCond and High > High[1];
	if BuyCond then begin
		Buy ("LE") OrderSize contracts next bar market;
	end;
end else begin
	if IsTXFSettlementDay and 1200 < Time and Time < 1400 then begin
		Sell ("LX_Settle") all contracts next bar market;
	end;

	vars: SellCond(True);
	SellCond = True;
	SellCond = SellCond and CountIf(Close < MA, 2 + 5) >= 5;
	SellCond = SellCond and CountIf(Close > MA, 5) > 0;
	SellCond = SellCond and Low < Low[1];
	if SellCond then begin
		Sell ("LX_Out") all contracts next bar market;
	end;

	vars: ATR(0);
	ATR = AvgTrueRange(30);
	SetStopLoss(5 * ATR * BigPointValue);
end;
